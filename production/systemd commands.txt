create service script : 	sudo nano /etc/systemd/system/syncRawData.service

reload daemon : 		sudo systemctl daemon-reload

enable service : 		sudo systemctl enable test.service

start service : 		sudo systemctl start test.service

stop service : 			sudo systemctl stop name_of_service

restart service : 		sudo systemctl restart name_of_service

check status : 			sudo systemctl status name_of_service

check journal :         sudo journalctl -u service-name.service --since "2 hours ago"
